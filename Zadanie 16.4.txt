SELECT I.DESCRIPTION, K.NAME
FROM ISSUES I
JOIN ISSUESLISTS K ON I.ISSUESLIST_ID = K.ID;

SELECT I.DESCRIPTION, U.FIRSTNAME
FROM ISSUES I
JOIN USERS U ON I.USER_ID_ASSIGNEDTO = U.ID;

SELECT U.FIRSTNAME, U.LASTNAME, COUNT(*) AS TASK_NUMBER
FROM USERS U JOIN ISSUES P ON U.ID = P.USER_ID_ASSIGNEDTO
GROUP BY P.USER_ID_ASSIGNEDTO
HAVING COUNT(*) > 0;
